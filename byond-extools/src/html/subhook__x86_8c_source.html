208  uint8_t *code = src;
209  size_t i;
210  int len = 0;
211  int operand_size = 4;
212  uint8_t opcode = 0;
213  int found_opcode = false;
214 
215  for (i = 0; i < sizeof(prefixes) / sizeof(*prefixes); i++) {
216  if (code[len] == prefixes[i]) {
217  len++;
218  if (prefixes[i] == 0x66) {
219  operand_size = 2;
220  }
221  }
222  }
223 
224 #ifdef SUBHOOK_X86_64
225  if ((code[len] & 0xF0) == 0x40) {
226  /* This is a REX prefix (40H - 4FH). REX prefixes are valid only in
227  * 64-bit mode.
228  */
229  uint8_t rex = code[len++];
230 
231  if (rex & 8) {
232  /* REX.W changes size of immediate operand to 64 bits. */
233  operand_size = 8;
234  }
235  }
236 #endif
237 
238  for (i = 0; i < sizeof(opcodes) / sizeof(*opcodes); i++) {
239  if (code[len] == opcodes[i].opcode) {
240  if (opcodes[i].flags & REG_OPCODE) {
241  found_opcode = ((code[len + 1] >> 3) & 7) == opcodes[i].reg_opcode;
242  } else {
243  found_opcode = true;
244  }
245  }
246 
247  if ((opcodes[i].flags & PLUS_R)
248  && (code[len] & 0xF8) == opcodes[i].opcode) {
249  found_opcode = true;
250  }
251 
252  if (found_opcode) {
253  opcode = code[len++];
254  break;
255  }
256  }
257 
258  if (!found_opcode) {
259  return 0;
260  }
261 
262  if (reloc_op_offset != NULL && opcodes[i].flags & RELOC) {
263  *reloc_op_offset = len; /* relative call or jump */
264  }
265 
266  if (opcodes[i].flags & MODRM) {
267  uint8_t modrm = code[len++]; /* +1 for Mod/RM byte */
268  uint8_t mod = modrm >> 6;
269  uint8_t rm = modrm & 7;
270 
271  if (mod != 3 && rm == 4) {
272  uint8_t sib = code[len++]; /* +1 for SIB byte */
273  uint8_t base = sib & 7;
274 
275  if (base == 5) {
276  /* The SIB is followed by a disp32 with no base if the MOD is 00B.
277  * Otherwise, disp8 or disp32 + [EBP].
278  */
279  if (mod == 1) {
280  len += 1; /* for disp8 */
281  } else {
282  len += 4; /* for disp32 */
283  }
284  }
285  }
286 
287 #ifdef SUBHOOK_X86_64
288  if (reloc_op_offset != NULL && rm == 5) {
289  *reloc_op_offset = (int32_t)len; /* RIP-relative addressing */
290  }
291 #endif
292 
293  if (mod == 1) {
294  len += 1; /* for disp8 */
295  }
296  if (mod == 2 || (mod == 0 && rm == 5)) {
297  len += 4; /* for disp32 */
298  }
299  }
300 
301  if (opcodes[i].flags & IMM8) {
302  len += 1;
303  }
304  if (opcodes[i].flags & IMM16) {
305  len += 2;
306  }
307  if (opcodes[i].flags & IMM32) {
308  len += operand_size;
309  }
310 
311  return len;
312 }

325 static int subhook_make_jmp32(void *src, void *dst) {
326  struct subhook_jmp32 *jmp = (struct subhook_jmp32 *)src;
327  intptr_t src_addr = (intptr_t)src;
328  intptr_t dst_addr = (intptr_t)dst;
329 #ifdef SUBHOOK_X86_64
330  int64_t distance = ABS(src_addr - dst_addr);
331 #endif
332 
333 #ifdef SUBHOOK_X86_64
334  if (distance < INT32_MIN || distance > INT32_MAX) {
335  return -EOVERFLOW;
336  }
337 #endif
338 
339  jmp->opcode = JMP_OPCODE;
340  jmp->offset = (int32_t)(dst_addr - (src_addr + sizeof(*jmp)));
341 
342  return 0;
343 }

377 static int subhook_make_trampoline(void *trampoline,
378  void *src,
379  size_t jmp_size,
380  size_t *trampoline_len,
381  subhook_flags_t flags) {
382  size_t orig_size = 0;
383  size_t insn_len;
384  intptr_t trampoline_addr = (intptr_t)trampoline;
385  intptr_t src_addr = (intptr_t)src;
386  subhook_disasm_handler_t disasm_handler =
387  subhook_disasm_handler != NULL ? subhook_disasm_handler : subhook_disasm;
388 
389  assert(trampoline_len != NULL);
390 
391  /* Determine how many bytes of original code needs to be copied over
392  * to the trampoline.
393  */
394  while (orig_size < jmp_size) {
395  int reloc_op_offset = 0;
396 
397  insn_len =
398  disasm_handler((void *)(src_addr + orig_size), &reloc_op_offset);
399 
400  if (insn_len == 0) {
401  return -EINVAL;
402  }
403 
404  /* Copy this instruction to the trampoline.
405  */
406  memcpy((void *)(trampoline_addr + orig_size),
407  (void *)(src_addr + orig_size),
408  insn_len);
409 
410  /* If the operand is a relative address, such as found in calls or
411  * jumps, it needs to be relocated because the original code and the
412  * trampoline reside at different locations in memory.
413  */
414  if (reloc_op_offset > 0) {
415  /* Calculate how far our trampoline is from the source and change
416  * the address accordingly.
417  */
418  int32_t offset = (int32_t)(trampoline_addr - src_addr);
419  int32_t *op = (int32_t *)(trampoline_addr + orig_size + reloc_op_offset);
420  *op -= offset;
421  }
422 
423  orig_size += insn_len;
424  }
425 
426  *trampoline_len = orig_size + jmp_size;
427 
428  /* Insert the final jump. It goes back to the original code at
429  * src + orig_size.
430  */
431  return subhook_make_jmp((void *)(trampoline_addr + orig_size),
432  (void *)(src_addr + orig_size),
433  flags);
434 }

437  void *dst,
438  subhook_flags_t flags) {
439  subhook_t hook;
440 
441  if ((hook = malloc(sizeof(*hook))) == NULL) {
442  return NULL;
443  }
444 
445  hook->installed = 0;
446  hook->src = src;
447  hook->dst = dst;
448  hook->flags = flags;
449  hook->jmp_size = subhook_get_jmp_size(hook->flags);
450  hook->trampoline_size = hook->jmp_size * 2 + MAX_INSN_LEN;
451  hook->trampoline_len = 0;
452 
453  if ((hook->code = malloc(hook->jmp_size)) == NULL) {
454  free(hook);
455  return NULL;
456  }
457 
458  memcpy(hook->code, hook->src, hook->jmp_size);
459 
460  if ((hook->trampoline = calloc(1, hook->trampoline_size)) == NULL) {
461  free(hook->code);
462  free(hook);
463  return NULL;
464  }
465 
466  if (subhook_unprotect(hook->src, hook->jmp_size) == NULL
467  || subhook_unprotect(hook->trampoline, hook->trampoline_size) == NULL)
468  {
469  free(hook->trampoline);
470  free(hook->code);
471  free(hook);
472  return NULL;
473  }
474 
475  subhook_make_trampoline(
476  hook->trampoline,
477  hook->src,
478  hook->jmp_size,
479  &hook->trampoline_len,
480  hook->flags);
481 
482  if (hook->trampoline_len == 0) {
483  free(hook->trampoline);
484  hook->trampoline = NULL;
485  }
486 
487  return hook;
488 }
